Build Autocomplete with altor-vec

What you'll build: A semantic typeahead that suggests the right intent even when the typed prefix does not literally match the stored label.

Bundle cost: 54KB · Query time: <1ms · Backend: None required

Install

npm install altor-vec

The Concept

Autocomplete feels best when the UI can recover from imperfect phrasing. Users type fragments like "forgot login" or "change email" and expect the right suggestion immediately, even when your stored labels use different wording.

Client-side vector search is a strong fit because suggestion lookups happen on every keystroke. Keeping the index local removes network jitter, keeps the interaction under a millisecond, and avoids sending half-typed queries to a third party.

This runnable example uses a deterministic text-to-vector helper so you can paste it into a project with only altor-vec installed. In production, replace that helper with real embeddings from Transformers.js or a precomputed pipeline.

This page uses a deterministic embedding helper so the snippets are runnable with only altor-vec. For production ranking quality, replace that helper with a real embedding model or precomputed vectors.

Full Code Example

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Reset password",
    "text": "Help users recover access after they forget a password or cannot sign in.",
    "meta": "account"
  },
  {
    "title": "Update billing card",
    "text": "Change the payment method for an active subscription or invoice.",
    "meta": "billing"
  },
  {
    "title": "Invite teammate",
    "text": "Add a coworker to the workspace with the right permissions.",
    "meta": "team"
  },
  {
    "title": "Export analytics report",
    "text": "Download weekly metrics as CSV or PDF for leadership review.",
    "meta": "reporting"
  },
  {
    "title": "Enable single sign-on",
    "text": "Set up SSO with SAML for enterprise logins.",
    "meta": "security"
  },
  {
    "title": "Change notification email",
    "text": "Update where invoices and alerts are delivered.",
    "meta": "account"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

async function main() {
  await init();

  const flat = new Float32Array(
    records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
  );

  const engine = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
  const hits = JSON.parse(engine.search(new Float32Array(embedText('forgot my password')), 4));

  const results = hits.map(([id, distance]) => ({
    ...records[id],
    similarity: Number((1 - distance).toFixed(3)),
  }));

  console.table(results);
  engine.free();
}

main();

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Reset password",
    "text": "Help users recover access after they forget a password or cannot sign in.",
    "meta": "account"
  },
  {
    "title": "Update billing card",
    "text": "Change the payment method for an active subscription or invoice.",
    "meta": "billing"
  },
  {
    "title": "Invite teammate",
    "text": "Add a coworker to the workspace with the right permissions.",
    "meta": "team"
  },
  {
    "title": "Export analytics report",
    "text": "Download weekly metrics as CSV or PDF for leadership review.",
    "meta": "reporting"
  },
  {
    "title": "Enable single sign-on",
    "text": "Set up SSO with SAML for enterprise logins.",
    "meta": "security"
  },
  {
    "title": "Change notification email",
    "text": "Update where invoices and alerts are delivered.",
    "meta": "account"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

export function AutocompleteExample() {
  const [engine, setEngine] = useState(null);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  useEffect(() => {
    let cancelled = false;
    let instance;

    (async () => {
      await init();
      const flat = new Float32Array(
        records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
      );
      instance = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
      if (!cancelled) setEngine(instance);
    })();

    return () => {
      cancelled = true;
      instance?.free();
    };
  }, []);

  useEffect(() => {
    if (!engine || query.trim().length < 2) {
      setResults([]);
      return;
    }

    const hits = JSON.parse(engine.search(new Float32Array(embedText(query)), 5));
    setResults(
      hits.map(([id, distance]) => ({
        ...records[id],
        similarity: Number((1 - distance).toFixed(3)),
      }))
    );
  }, [engine, query]);

  return (
    <section>
      <input
        value={query}
        onChange={(event) => setQuery(event.target.value)}
        placeholder="Type a support question"
      />
      <ul>
        {results.map((result) => (
          <li key={result.title}>
            <strong>{result.title}</strong> — {result.meta} (score {result.similarity})
          </li>
        ))}
      </ul>
    </section>
  );
}

How It Works

The flow is the same pattern you would use in production: turn each record into an embedding vector, build an HNSW graph with WasmSearchEngine.from_vectors(), and query it with the embedding for the current search text. HNSW keeps approximate nearest-neighbor search fast by traversing a layered graph instead of scanning every vector linearly.

Similarity scoring here is derived from the returned distance with 1 - distance so the UI can show an intuitive confidence number. If you swap in a real embedding model such as MiniLM, multilingual-e5, or precomputed server-generated vectors, the surrounding indexing and querying code stays almost identical.

When to Use This Pattern

  • Command palettes and app launchers
  • Support forms that should suggest likely intents
  • Docs search boxes with semantic typeahead
  • Mobile UIs where every network roundtrip is noticeable

Limitations

The demo embedding function is intentionally lightweight, so ranking quality is limited compared with a real model. Large suggestion sets still consume browser memory, and truly multilingual autocomplete needs embeddings trained across languages.
