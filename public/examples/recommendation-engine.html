Build Recommendation Engine with altor-vec

What you'll build: A related-items rail that recommends nearby products or articles from the current item embedding instead of hand-written rules.

Bundle cost: 54KB · Query time: <1ms · Backend: None required

Install

npm install altor-vec

The Concept

Recommendation engines often start with categories and tags, but those rules miss nuance. Two items can be related because they solve the same job, share tone, or fit the same usage context even if the taxonomy is different.

Running the nearest-neighbor lookup in the browser works well for small and medium catalogs, especially when the recommendation request happens on page load or after a user opens a detail view. The user gets instant related results without waiting on a ranking API.

The example below builds an HNSW index in memory and uses the current item description as the query vector. It is real JavaScript you can run today, then swap to better embeddings once the pattern is proven.

This page uses a deterministic embedding helper so the snippets are runnable with only altor-vec. For production ranking quality, replace that helper with a real embedding model or precomputed vectors.

Full Code Example

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Compact mechanical keyboard",
    "text": "Portable low-profile keyboard designed for frequent travel and small desks.",
    "meta": "peripherals"
  },
  {
    "title": "Silent wireless mouse",
    "text": "Quiet click mouse for shared offices and late-night work.",
    "meta": "peripherals"
  },
  {
    "title": "USB-C travel dock",
    "text": "Slim dock with HDMI, charging, and SD card support for laptop setups.",
    "meta": "accessories"
  },
  {
    "title": "Noise-isolating headset",
    "text": "Lightweight headset for calls in busy environments.",
    "meta": "audio"
  },
  {
    "title": "Laptop stand",
    "text": "Foldable aluminum stand that fits inside a backpack.",
    "meta": "ergonomics"
  },
  {
    "title": "Portable monitor",
    "text": "Thin second screen for working from hotel rooms or client sites.",
    "meta": "displays"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

async function main() {
  await init();

  const flat = new Float32Array(
    records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
  );

  const engine = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
  const hits = JSON.parse(engine.search(new Float32Array(embedText('quiet travel keyboard')), 4));

  const results = hits.map(([id, distance]) => ({
    ...records[id],
    similarity: Number((1 - distance).toFixed(3)),
  }));

  console.table(results);
  engine.free();
}

main();

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Compact mechanical keyboard",
    "text": "Portable low-profile keyboard designed for frequent travel and small desks.",
    "meta": "peripherals"
  },
  {
    "title": "Silent wireless mouse",
    "text": "Quiet click mouse for shared offices and late-night work.",
    "meta": "peripherals"
  },
  {
    "title": "USB-C travel dock",
    "text": "Slim dock with HDMI, charging, and SD card support for laptop setups.",
    "meta": "accessories"
  },
  {
    "title": "Noise-isolating headset",
    "text": "Lightweight headset for calls in busy environments.",
    "meta": "audio"
  },
  {
    "title": "Laptop stand",
    "text": "Foldable aluminum stand that fits inside a backpack.",
    "meta": "ergonomics"
  },
  {
    "title": "Portable monitor",
    "text": "Thin second screen for working from hotel rooms or client sites.",
    "meta": "displays"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

export function RecommendationEngineExample() {
  const [engine, setEngine] = useState(null);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  useEffect(() => {
    let cancelled = false;
    let instance;

    (async () => {
      await init();
      const flat = new Float32Array(
        records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
      );
      instance = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
      if (!cancelled) setEngine(instance);
    })();

    return () => {
      cancelled = true;
      instance?.free();
    };
  }, []);

  useEffect(() => {
    if (!engine || query.trim().length < 2) {
      setResults([]);
      return;
    }

    const hits = JSON.parse(engine.search(new Float32Array(embedText(query)), 5));
    setResults(
      hits.map(([id, distance]) => ({
        ...records[id],
        similarity: Number((1 - distance).toFixed(3)),
      }))
    );
  }, [engine, query]);

  return (
    <section>
      <input
        value={query}
        onChange={(event) => setQuery(event.target.value)}
        placeholder="Describe the current item"
      />
      <ul>
        {results.map((result) => (
          <li key={result.title}>
            <strong>{result.title}</strong> — {result.meta} (score {result.similarity})
          </li>
        ))}
      </ul>
    </section>
  );
}

How It Works

The flow is the same pattern you would use in production: turn each record into an embedding vector, build an HNSW graph with WasmSearchEngine.from_vectors(), and query it with the embedding for the current search text. HNSW keeps approximate nearest-neighbor search fast by traversing a layered graph instead of scanning every vector linearly.

Similarity scoring here is derived from the returned distance with 1 - distance so the UI can show an intuitive confidence number. If you swap in a real embedding model such as MiniLM, multilingual-e5, or precomputed server-generated vectors, the surrounding indexing and querying code stays almost identical.

When to Use This Pattern

  • Related product modules on ecommerce detail pages
  • Newsletter or article recommendations in content apps
  • Upsell surfaces inside desktop software
  • Personalization experiments without adding backend ranking infrastructure

Limitations

For very large catalogs or collaborative filtering signals, server-side recommenders still win. You also need consistent embeddings for both catalog items and the current context, and full personalization across users usually requires backend state.
