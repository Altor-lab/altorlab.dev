Build Code Search with altor-vec

What you'll build: A code snippet search that returns the most relevant helper, hook, or utility from local repository metadata.

Bundle cost: 54KB · Query time: <1ms · Backend: None required

Install

npm install altor-vec

The Concept

Developers often remember what a piece of code does, not where it lives or the exact function name. Semantic vectors are useful because they connect problem descriptions like "parse auth token" to the relevant snippet metadata.

Client-side code search works well for demos, small repos, documentation examples, and packaged desktop tools. It is also a good complement to exact symbol search when the user knows the intent but not the identifier.

The runnable example below indexes short snippet descriptions so you can test the retrieval loop without extra services. In a real codebase, generate embeddings from comments, docstrings, symbol names, and surrounding context.

This page uses a deterministic embedding helper so the snippets are runnable with only altor-vec. For production ranking quality, replace that helper with a real embedding model or precomputed vectors.

Full Code Example

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "useDebouncedValue",
    "text": "React hook that delays updates for search inputs and filters.",
    "meta": "react-hooks.ts"
  },
  {
    "title": "fetchWithRetry",
    "text": "Utility for retrying transient HTTP failures with exponential backoff.",
    "meta": "network.ts"
  },
  {
    "title": "parseJwt",
    "text": "Decode token payloads and check expiry for client auth flows.",
    "meta": "auth.ts"
  },
  {
    "title": "buildCsv",
    "text": "Serialize table rows into downloadable CSV output.",
    "meta": "export.ts"
  },
  {
    "title": "groupByDay",
    "text": "Aggregate events into daily buckets for chart rendering.",
    "meta": "analytics.ts"
  },
  {
    "title": "useKeyboardShortcut",
    "text": "Register global shortcuts for command palette interactions.",
    "meta": "shortcuts.ts"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

async function main() {
  await init();

  const flat = new Float32Array(
    records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
  );

  const engine = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
  const hits = JSON.parse(engine.search(new Float32Array(embedText('debounced search input hook')), 4));

  const results = hits.map(([id, distance]) => ({
    ...records[id],
    similarity: Number((1 - distance).toFixed(3)),
  }));

  console.table(results);
  engine.free();
}

main();

    "text": "Aggregate events into daily buckets for chart rendering.",

  },
  {
    "title": "useKeyboardShortcut",
    "text": "Register global shortcuts for command palette interactions.",
    "meta": "shortcuts.ts"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

export function CodeSearchExample() {
  const [engine, setEngine] = useState(null);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  useEffect(() => {
    let cancelled = false;
    let instance;

    (async () => {
      await init();
      const flat = new Float32Array(
        records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
      );
      instance = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
      if (!cancelled) setEngine(instance);
    })();

    return () => {
      cancelled = true;
      instance?.free();
    };
  }, []);

  useEffect(() => {
    if (!engine || query.trim().length < 2) {
      setResults([]);
      return;
    }

    const hits = JSON.parse(engine.search(new Float32Array(embedText(query)), 5));
    setResults(
      hits.map(([id, distance]) => ({
        ...records[id],
        similarity: Number((1 - distance).toFixed(3)),
      }))
    );
  }, [engine, query]);

  return (
    <section>
      <input
        value={query}
        onChange={(event) => setQuery(event.target.value)}
        placeholder="Describe the code you need"
      />
      <ul>
        {results.map((result) => (
          <li key={result.title}>
            <strong>{result.title}</strong> — {result.meta} (score {result.similarity})
          </li>
        ))}
      </ul>
    </section>
  );
}

How It Works

The flow is the same pattern you would use in production: turn each record into an embedding vector, build an HNSW graph with WasmSearchEngine.from_vectors(), and query it with the embedding for the current search text. HNSW keeps approximate nearest-neighbor search fast by traversing a layered graph instead of scanning every vector linearly.

Similarity scoring here is derived from the returned distance with 1 - distance so the UI can show an intuitive confidence number. If you swap in a real embedding model such as MiniLM, multilingual-e5, or precomputed server-generated vectors, the surrounding indexing and querying code stays almost identical.

When to Use This Pattern

  • Snippet libraries and internal code recipes
  • Docs sites that embed runnable examples
  • IDE extensions with a local semantic palette
  • Repository knowledge tools for onboarding developers

Limitations

Pure semantic search should not replace syntax-aware or exact symbol search. Large repositories need a preprocessing pipeline, and browser delivery may expose source metadata that should stay private.
