Build Chat Memory with altor-vec

What you'll build: A semantic memory layer that recalls the most relevant prior chat facts for the current turn.

Bundle cost: 54KB · Query time: <1ms · Backend: None required

Install

npm install altor-vec

The Concept

Chat memory gets noisy if you only keep the last N messages. Semantic retrieval helps you recall the specific prior fact that matters now, such as a deadline, product preference, or issue summary mentioned twenty turns ago.

A local memory index is useful for copilots and assistants that run in the browser because it keeps user conversation state on-device. Queries stay fast and the assistant can recover context without sending the full transcript elsewhere.

The sample code indexes short memory snippets and then queries them like any other vector corpus. In production, store timestamps, recency weights, and metadata so you can combine semantic relevance with conversation rules.

This page uses a deterministic embedding helper so the snippets are runnable with only altor-vec. For production ranking quality, replace that helper with a real embedding model or precomputed vectors.

Full Code Example

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Deployment date",
    "text": "Customer plans production rollout on April 14 after final security review.",
    "meta": "timeline"
  },
  {
    "title": "Preferred cloud region",
    "text": "EU hosting is required for the first enterprise deployment.",
    "meta": "infra"
  },
  {
    "title": "Renewal risk",
    "text": "Account is blocked on SSO support and audit log exports.",
    "meta": "risk"
  },
  {
    "title": "Success metric",
    "text": "Team wants search latency under 20 milliseconds for agent workflows.",
    "meta": "goal"
  },
  {
    "title": "Meeting follow-up",
    "text": "Send architecture diagram and pricing comparison by Friday.",
    "meta": "task"
  },
  {
    "title": "Security note",
    "text": "Customer does not allow raw support queries to leave the browser.",
    "meta": "security"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

async function main() {
  await init();

  const flat = new Float32Array(
    records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
  );

  const engine = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
  const hits = JSON.parse(engine.search(new Float32Array(embedText('what was the customer deployment date')), 4));

  const results = hits.map(([id, distance]) => ({
    ...records[id],
    similarity: Number((1 - distance).toFixed(3)),
  }));

  console.table(results);
  engine.free();
}

main();

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Deployment date",
    "text": "Customer plans production rollout on April 14 after final security review.",
    "meta": "timeline"
  },
  {
    "title": "Preferred cloud region",
    "text": "EU hosting is required for the first enterprise deployment.",
    "meta": "infra"
  },
  {
    "title": "Renewal risk",
    "text": "Account is blocked on SSO support and audit log exports.",
    "meta": "risk"
  },
  {
    "title": "Success metric",
    "text": "Team wants search latency under 20 milliseconds for agent workflows.",
    "meta": "goal"
  },
  {
    "title": "Meeting follow-up",
    "text": "Send architecture diagram and pricing comparison by Friday.",
    "meta": "task"
  },
  {
    "title": "Security note",
    "text": "Customer does not allow raw support queries to leave the browser.",
    "meta": "security"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

export function ChatMemoryExample() {
  const [engine, setEngine] = useState(null);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  useEffect(() => {
    let cancelled = false;
    let instance;

    (async () => {
      await init();
      const flat = new Float32Array(
        records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
      );
      instance = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
      if (!cancelled) setEngine(instance);
    })();

    return () => {
      cancelled = true;
      instance?.free();
    };
  }, []);

  useEffect(() => {
    if (!engine || query.trim().length < 2) {
      setResults([]);
      return;
    }

    const hits = JSON.parse(engine.search(new Float32Array(embedText(query)), 5));
    setResults(
      hits.map(([id, distance]) => ({
        ...records[id],
        similarity: Number((1 - distance).toFixed(3)),
      }))
    );
  }, [engine, query]);

  return (
    <section>
      <input
        value={query}
        onChange={(event) => setQuery(event.target.value)}
        placeholder="Ask a follow-up"
      />
      <ul>
        {results.map((result) => (
          <li key={result.title}>
            <strong>{result.title}</strong> — {result.meta} (score {result.similarity})
          </li>
        ))}
      </ul>
    </section>
  );
}

How It Works

The flow is the same pattern you would use in production: turn each record into an embedding vector, build an HNSW graph with WasmSearchEngine.from_vectors(), and query it with the embedding for the current search text. HNSW keeps approximate nearest-neighbor search fast by traversing a layered graph instead of scanning every vector linearly.

Similarity scoring here is derived from the returned distance with 1 - distance so the UI can show an intuitive confidence number. If you swap in a real embedding model such as MiniLM, multilingual-e5, or precomputed server-generated vectors, the surrounding indexing and querying code stays almost identical.

When to Use This Pattern

  • Personal AI assistants
  • Customer success copilots inside a CRM
  • Local note-taking chat interfaces
  • Browser extensions that need short-term memory without a backend

Limitations

Memory retrieval still requires policies for what should be stored, when it should expire, and how to handle conflicting facts. Sensitive conversations may also need encryption and explicit user controls around retention.
