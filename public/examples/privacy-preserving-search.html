Build Privacy-Preserving Search with altor-vec

What you'll build: A private search surface for sensitive notes or policies where the query vector and index stay on the device.

Bundle cost: 54KB · Query time: <1ms · Backend: None required

Install

npm install altor-vec

The Concept

Some search surfaces carry sensitive intent: legal notes, health content, internal strategy docs, or customer records. Sending every query to a hosted vector API can create governance and audit overhead that is unnecessary for moderate datasets.

Client-side vector search keeps retrieval local. That does not solve every privacy problem, but it does mean the search string, embedding, and nearest-neighbor computation can stay inside the browser session.

The example uses a local HNSW index built at runtime for clarity. Production systems often ship a prebuilt encrypted or access-controlled index, then layer analytics carefully so raw queries are not leaked through telemetry.

This page uses a deterministic embedding helper so the snippets are runnable with only altor-vec. For production ranking quality, replace that helper with a real embedding model or precomputed vectors.

Full Code Example

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Data retention clause",
    "text": "Defines how long customer account and billing records remain stored.",
    "meta": "legal"
  },
  {
    "title": "Subprocessor disclosure",
    "text": "Lists third-party vendors involved in hosting, analytics, and email.",
    "meta": "privacy"
  },
  {
    "title": "Security incident notice",
    "text": "Timeline and communication requirements after a reportable breach.",
    "meta": "security"
  },
  {
    "title": "DPA template",
    "text": "Data processing addendum terms for enterprise contracts.",
    "meta": "compliance"
  },
  {
    "title": "Employee access policy",
    "text": "Rules for privileged access and periodic review.",
    "meta": "internal"
  },
  {
    "title": "Regional deletion schedule",
    "text": "Country-specific retention and deletion windows for customer data.",
    "meta": "privacy"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

async function main() {
  await init();

  const flat = new Float32Array(
    records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
  );

  const engine = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
  const hits = JSON.parse(engine.search(new Float32Array(embedText('contract clause about data retention')), 4));

  const results = hits.map(([id, distance]) => ({
    ...records[id],
    similarity: Number((1 - distance).toFixed(3)),
  }));

  console.table(results);
  engine.free();
}

main();

    "text": "Data processing addendum terms for enterprise contracts.",
    "meta": "compliance"
  },
  {
    "title": "Employee access policy",
    "text": "Rules for privileged access and periodic review.",
    "meta": "internal"
  },
  {
    "title": "Regional deletion schedule",
    "text": "Country-specific retention and deletion windows for customer data.",
    "meta": "privacy"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

export function PrivacyPreservingSearchExample() {
  const [engine, setEngine] = useState(null);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  useEffect(() => {
    let cancelled = false;
    let instance;

    (async () => {
      await init();
      const flat = new Float32Array(
        records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
      );
      instance = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
      if (!cancelled) setEngine(instance);
    })();

    return () => {
      cancelled = true;
      instance?.free();
    };
  }, []);

  useEffect(() => {
    if (!engine || query.trim().length < 2) {
      setResults([]);
      return;
    }

    const hits = JSON.parse(engine.search(new Float32Array(embedText(query)), 5));
    setResults(
      hits.map(([id, distance]) => ({
        ...records[id],
        similarity: Number((1 - distance).toFixed(3)),
      }))
    );
  }, [engine, query]);

  return (
    <section>
      <input
        value={query}
        onChange={(event) => setQuery(event.target.value)}
        placeholder="Search locally"
      />
      <ul>
        {results.map((result) => (
          <li key={result.title}>
            <strong>{result.title}</strong> — {result.meta} (score {result.similarity})
          </li>
        ))}
      </ul>
    </section>
  );
}

How It Works

The flow is the same pattern you would use in production: turn each record into an embedding vector, build an HNSW graph with WasmSearchEngine.from_vectors(), and query it with the embedding for the current search text. HNSW keeps approximate nearest-neighbor search fast by traversing a layered graph instead of scanning every vector linearly.

Similarity scoring here is derived from the returned distance with 1 - distance so the UI can show an intuitive confidence number. If you swap in a real embedding model such as MiniLM, multilingual-e5, or precomputed server-generated vectors, the surrounding indexing and querying code stays almost identical.

When to Use This Pattern

  • Employee policy or legal search portals
  • Healthcare and finance note viewers
  • Personal knowledge management apps
  • Security-conscious enterprise desktop apps

Limitations

If the browser downloads the underlying content, users can inspect it, so this pattern is not a substitute for access control. Sensitive telemetry, crash reporting, or third-party scripts can also leak query data if you do not audit them carefully.
