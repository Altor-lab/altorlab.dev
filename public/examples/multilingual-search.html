Build Multilingual Search with altor-vec

What you'll build: A cross-language search UI that lets a Spanish or French query retrieve the right English support article.

Bundle cost: 54KB · Query time: <1ms · Backend: None required

Install

npm install altor-vec

The Concept

Multilingual search matters when your content language and your user language do not match. With aligned embeddings, a Spanish question can retrieve the best English article without machine-translating every page or maintaining duplicate taxonomies.

Running the lookup locally is useful for product docs, support widgets, and mobile apps where you want the language bridge without another network dependency. The browser only needs the embeddings and metadata.

The demo keeps the code runnable by using a deterministic helper and multilingual sample phrases. In production, use multilingual embeddings so semantically equivalent queries from different languages land near each other.

This page uses a deterministic embedding helper so the snippets are runnable with only altor-vec. For production ranking quality, replace that helper with a real embedding model or precomputed vectors.

Full Code Example

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Update payment method",
    "text": "Change the saved card used for your subscription invoices.",
    "meta": "english"
  },
  {
    "title": "Reset your password",
    "text": "Recover access when you cannot sign in to the workspace.",
    "meta": "english"
  },
  {
    "title": "Enable single sign-on",
    "text": "Configure SAML-based login for enterprise accounts.",
    "meta": "english"
  },
  {
    "title": "T\u00e9l\u00e9charger les factures",
    "text": "Exporter invoice PDFs from the billing history page.",
    "meta": "fran\u00e7ais"
  },
  {
    "title": "Invitar compa\u00f1eros",
    "text": "Agregar nuevos usuarios al espacio de trabajo con permisos.",
    "meta": "espa\u00f1ol"
  },
  {
    "title": "Configurar alertas",
    "text": "Create email notifications for incidents and usage thresholds.",
    "meta": "espa\u00f1ol"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

async function main() {
  await init();

  const flat = new Float32Array(
    records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
  );

  const engine = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
  const hits = JSON.parse(engine.search(new Float32Array(embedText('cambiar tarjeta de pago')), 4));

  const results = hits.map(([id, distance]) => ({
    ...records[id],
    similarity: Number((1 - distance).toFixed(3)),
  }));

  console.table(results);
  engine.free();
}

main();

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Update payment method",
    "text": "Change the saved card used for your subscription invoices.",
    "meta": "english"
  },
  {
    "title": "Reset your password",
    "text": "Recover access when you cannot sign in to the workspace.",
    "meta": "english"
  },
  {
    "title": "Enable single sign-on",
    "text": "Configure SAML-based login for enterprise accounts.",
    "meta": "english"
  },
  {
    "title": "T\u00e9l\u00e9charger les factures",
    "text": "Exporter invoice PDFs from the billing history page.",
    "meta": "fran\u00e7ais"
  },
  {
    "title": "Invitar compa\u00f1eros",
    "text": "Agregar nuevos usuarios al espacio de trabajo con permisos.",
    "meta": "espa\u00f1ol"
  },
  {
    "title": "Configurar alertas",
    "text": "Create email notifications for incidents and usage thresholds.",
    "meta": "espa\u00f1ol"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

export function MultilingualSearchExample() {
  const [engine, setEngine] = useState(null);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  useEffect(() => {
    let cancelled = false;
    let instance;

    (async () => {
      await init();
      const flat = new Float32Array(
        records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
      );
      instance = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
      if (!cancelled) setEngine(instance);
    })();

    return () => {
      cancelled = true;
      instance?.free();
    };
  }, []);

  useEffect(() => {
    if (!engine || query.trim().length < 2) {
      setResults([]);
      return;
    }

    const hits = JSON.parse(engine.search(new Float32Array(embedText(query)), 5));
    setResults(
      hits.map(([id, distance]) => ({
        ...records[id],
        similarity: Number((1 - distance).toFixed(3)),
      }))
    );
  }, [engine, query]);

  return (
    <section>
      <input
        value={query}
        onChange={(event) => setQuery(event.target.value)}
        placeholder="Search in any supported language"
      />
      <ul>
        {results.map((result) => (
          <li key={result.title}>
            <strong>{result.title}</strong> — {result.meta} (score {result.similarity})
          </li>
        ))}
      </ul>
    </section>
  );
}

How It Works

The flow is the same pattern you would use in production: turn each record into an embedding vector, build an HNSW graph with WasmSearchEngine.from_vectors(), and query it with the embedding for the current search text. HNSW keeps approximate nearest-neighbor search fast by traversing a layered graph instead of scanning every vector linearly.

Similarity scoring here is derived from the returned distance with 1 - distance so the UI can show an intuitive confidence number. If you swap in a real embedding model such as MiniLM, multilingual-e5, or precomputed server-generated vectors, the surrounding indexing and querying code stays almost identical.

When to Use This Pattern

  • Global SaaS help centers
  • Product docs with English source content and international audiences
  • Mobile apps serving multilingual support
  • Internal tools across distributed global teams

Limitations

Real multilingual quality depends entirely on the embedding model. Without aligned multilingual embeddings, cross-language retrieval quality will drop sharply, and language-specific synonyms still need evaluation with native speakers.
