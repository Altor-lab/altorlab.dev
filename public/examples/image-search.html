Build Image Search with altor-vec

What you'll build: An image finder that searches caption embeddings locally so users can type concepts like "sunset over city skyline" and get visual matches.

Bundle cost: 54KB · Query time: <1ms · Backend: None required

Install

npm install altor-vec

The Concept

Vector search is a practical way to power image discovery when you have captions, tags, or multimodal embeddings. Users describe the scene they want instead of memorizing filenames or exact labels.

A browser-side index works well for curated libraries, portfolios, or offline media tools. Once assets and metadata are cached, searches remain instant and private.

The example keeps the embedding step simple by encoding text descriptions directly. Swap the helper for a caption embedding model or multimodal embeddings when you connect this pattern to real images.

This page uses a deterministic embedding helper so the snippets are runnable with only altor-vec. For production ranking quality, replace that helper with a real embedding model or precomputed vectors.

Full Code Example

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Golden hour skyline",
    "text": "Downtown skyline shot with orange sunset light and reflective windows.",
    "meta": "city"
  },
  {
    "title": "Forest waterfall",
    "text": "Long-exposure image of a waterfall surrounded by moss and trees.",
    "meta": "nature"
  },
  {
    "title": "Night market",
    "text": "Street food stalls with neon lights and a crowded city scene.",
    "meta": "travel"
  },
  {
    "title": "Desert road",
    "text": "Open highway cutting through red rock desert at sunrise.",
    "meta": "landscape"
  },
  {
    "title": "Mountain lake",
    "text": "Calm alpine lake with snowy peaks mirrored in the water.",
    "meta": "nature"
  },
  {
    "title": "Studio portrait",
    "text": "High-contrast portrait with dark background and rim lighting.",
    "meta": "people"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

async function main() {
  await init();

  const flat = new Float32Array(
    records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
  );

  const engine = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
  const hits = JSON.parse(engine.search(new Float32Array(embedText('sunset over city skyline')), 4));

  const results = hits.map(([id, distance]) => ({
    ...records[id],
    similarity: Number((1 - distance).toFixed(3)),
  }));

  console.table(results);
  engine.free();
}

main();

import { useEffect, useState } from 'react';
import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Golden hour skyline",
    "text": "Downtown skyline shot with orange sunset light and reflective windows.",
    "meta": "city"
  },
  {
    "title": "Forest waterfall",
    "text": "Long-exposure image of a waterfall surrounded by moss and trees.",
    "meta": "nature"
  },
  {
    "title": "Night market",
    "text": "Street food stalls with neon lights and a crowded city scene.",
    "meta": "travel"
  },
  {
    "title": "Desert road",
    "text": "Open highway cutting through red rock desert at sunrise.",
    "meta": "landscape"
  },
  {
    "title": "Mountain lake",
    "text": "Calm alpine lake with snowy peaks mirrored in the water.",
    "meta": "nature"
  },
  {
    "title": "Studio portrait",
    "text": "High-contrast portrait with dark background and rim lighting.",
    "meta": "people"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

export function ImageSearchExample() {
  const [engine, setEngine] = useState(null);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  useEffect(() => {
    let cancelled = false;
    let instance;

    (async () => {
      await init();
      const flat = new Float32Array(
        records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
      );
      instance = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
      if (!cancelled) setEngine(instance);
    })();

    return () => {
      cancelled = true;
      instance?.free();
    };
  }, []);

  useEffect(() => {
    if (!engine || query.trim().length < 2) {
      setResults([]);
      return;
    }

    const hits = JSON.parse(engine.search(new Float32Array(embedText(query)), 5));
    setResults(
      hits.map(([id, distance]) => ({
        ...records[id],
        similarity: Number((1 - distance).toFixed(3)),
      }))
    );
  }, [engine, query]);

  return (
    <section>
      <input
        value={query}
        onChange={(event) => setQuery(event.target.value)}
        placeholder="Describe an image"
      />
      <ul>
        {results.map((result) => (
          <li key={result.title}>
            <strong>{result.title}</strong> — {result.meta} (score {result.similarity})
          </li>
        ))}
      </ul>
    </section>
  );
}

How It Works

The flow is the same pattern you would use in production: turn each record into an embedding vector, build an HNSW graph with WasmSearchEngine.from_vectors(), and query it with the embedding for the current search text. HNSW keeps approximate nearest-neighbor search fast by traversing a layered graph instead of scanning every vector linearly.

Similarity scoring here is derived from the returned distance with 1 - distance so the UI can show an intuitive confidence number. If you swap in a real embedding model such as MiniLM, multilingual-e5, or precomputed server-generated vectors, the surrounding indexing and querying code stays almost identical.

When to Use This Pattern

  • Photography portfolio search
  • Design asset pickers in creative tools
  • Offline museum or kiosk guides
  • Stock image browsers with small curated collections

Limitations

Real image search quality depends on strong captions or image embeddings. Large image libraries can make download size expensive, and multimodal models are heavier than the toy text helper used in this demo.
