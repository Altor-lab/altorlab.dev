Build FAQ Search with altor-vec

What you'll build: A help-center lookup that routes natural-language questions to the closest FAQ answer without a server.

Bundle cost: 54KB · Query time: <1ms · Backend: None required

Install

npm install altor-vec

The Concept

FAQs are a classic semantic search problem because users ask the same question in dozens of ways. A vector index lets "change payment card" and "update billing details" converge on the same answer.

Doing the lookup in the browser is ideal for support surfaces that need instant response and lower operational complexity. You avoid an API hop and can serve static help content directly from a CDN.

This copy-paste example indexes question-answer pairs in memory. It is enough to validate the UX before you move to precomputed embeddings for a larger production help center.

This page uses a deterministic embedding helper so the snippets are runnable with only altor-vec. For production ranking quality, replace that helper with a real embedding model or precomputed vectors.

Full Code Example

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "How do I reset my password?",
    "text": "Use the forgot password link on the sign-in page to receive a reset email.",
    "meta": "account"
  },
  {
    "title": "How do I update my billing card?",
    "text": "Open Billing, choose Payment method, and replace the saved card.",
    "meta": "billing"
  },
  {
    "title": "Can I invite teammates?",
    "text": "Owners and admins can invite users from the Team settings page.",
    "meta": "team"
  },
  {
    "title": "How do I download invoices?",
    "text": "Invoices are available as PDFs in Billing history.",
    "meta": "billing"
  },
  {
    "title": "Where do I enable SSO?",
    "text": "Enterprise customers can configure SAML under Security settings.",
    "meta": "security"
  },
  {
    "title": "How do I cancel my plan?",
    "text": "You can cancel or downgrade at the end of the current billing period.",
    "meta": "billing"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

async function main() {
  await init();

  const flat = new Float32Array(
    records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
  );

  const engine = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
  const hits = JSON.parse(engine.search(new Float32Array(embedText('can i change the card on my subscription')), 4));

  const results = hits.map(([id, distance]) => ({
    ...records[id],
    similarity: Number((1 - distance).toFixed(3)),
  }));

  console.table(results);
  engine.free();
}

main();

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "How do I reset my password?",
    "text": "Use the forgot password link on the sign-in page to receive a reset email.",
    "meta": "account"
  },
  {
    "title": "How do I update my billing card?",
    "text": "Open Billing, choose Payment method, and replace the saved card.",
    "meta": "billing"
  },
  {
    "title": "Can I invite teammates?",
    "text": "Owners and admins can invite users from the Team settings page.",
    "meta": "team"
  },
  {
    "title": "How do I download invoices?",
    "text": "Invoices are available as PDFs in Billing history.",
    "meta": "billing"
  },
  {
    "title": "Where do I enable SSO?",
    "text": "Enterprise customers can configure SAML under Security settings.",
    "meta": "security"
  },
  {
    "title": "How do I cancel my plan?",
    "text": "You can cancel or downgrade at the end of the current billing period.",
    "meta": "billing"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

export function FAQSearchExample() {
  const [engine, setEngine] = useState(null);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  useEffect(() => {
    let cancelled = false;
    let instance;

    (async () => {
      await init();
      const flat = new Float32Array(
        records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
      );
      instance = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
      if (!cancelled) setEngine(instance);
    })();

    return () => {
      cancelled = true;
      instance?.free();
    };
  }, []);

  useEffect(() => {
    if (!engine || query.trim().length < 2) {
      setResults([]);
      return;
    }

    const hits = JSON.parse(engine.search(new Float32Array(embedText(query)), 5));
    setResults(
      hits.map(([id, distance]) => ({
        ...records[id],
        similarity: Number((1 - distance).toFixed(3)),
      }))
    );
  }, [engine, query]);

  return (
    <section>
      <input
        value={query}
        onChange={(event) => setQuery(event.target.value)}
        placeholder="Ask a support question"
      />
      <ul>
        {results.map((result) => (
          <li key={result.title}>
            <strong>{result.title}</strong> — {result.meta} (score {result.similarity})
          </li>
        ))}
      </ul>
    </section>
  );
}

How It Works

The flow is the same pattern you would use in production: turn each record into an embedding vector, build an HNSW graph with WasmSearchEngine.from_vectors(), and query it with the embedding for the current search text. HNSW keeps approximate nearest-neighbor search fast by traversing a layered graph instead of scanning every vector linearly.

Similarity scoring here is derived from the returned distance with 1 - distance so the UI can show an intuitive confidence number. If you swap in a real embedding model such as MiniLM, multilingual-e5, or precomputed server-generated vectors, the surrounding indexing and querying code stays almost identical.

When to Use This Pattern

  • SaaS help widgets
  • Embedded support centers inside authenticated apps
  • Static support portals distributed globally via CDN
  • Agent assist tools that need a fast FAQ fallback

Limitations

FAQ search is only as good as the answer corpus. Exact policy references and account-specific questions may still need lexical search or backend logic, and larger help centers benefit from offline index generation.
