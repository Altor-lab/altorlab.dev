Build Similar Items with altor-vec

What you'll build: A "more like this" module that recommends the nearest items based on vector similarity.

Bundle cost: 54KB · Query time: <1ms · Backend: None required

Install

npm install altor-vec

The Concept

"Similar items" is one of the cleanest vector search patterns because the query is just another item embedding. Instead of manually curating pairs, you let semantic similarity discover products, articles, or assets with overlapping intent and style.

Client-side search is a good fit when the similar-items module runs on pages that already have the catalog metadata locally. The results arrive instantly and the page does not need another backend request to fill the recommendation rail.

This example uses short product summaries so the entire setup is runnable in one file. In production, you would index richer catalog embeddings created offline from descriptions, tags, and images.

This page uses a deterministic embedding helper so the snippets are runnable with only altor-vec. For production ranking quality, replace that helper with a real embedding model or precomputed vectors.

Full Code Example

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Everyday commuter backpack",
    "text": "Minimal black backpack with laptop sleeve and weather-resistant fabric.",
    "meta": "bags"
  },
  {
    "title": "Trail hydration pack",
    "text": "Running-focused pack with chest straps and soft flask pockets.",
    "meta": "bags"
  },
  {
    "title": "Weekend duffel",
    "text": "Large carry-all bag for short trips and gym gear.",
    "meta": "bags"
  },
  {
    "title": "Leather briefcase",
    "text": "Structured office bag with premium materials and organizer pockets.",
    "meta": "bags"
  },
  {
    "title": "Packable sling",
    "text": "Small cross-body bag for essentials during city travel.",
    "meta": "bags"
  },
  {
    "title": "Camera cube insert",
    "text": "Protective insert for carrying lenses inside a larger backpack.",
    "meta": "accessories"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

async function main() {
  await init();

  const flat = new Float32Array(
    records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
  );

  const engine = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
  const hits = JSON.parse(engine.search(new Float32Array(embedText('minimal everyday backpack')), 4));

  const results = hits.map(([id, distance]) => ({
    ...records[id],
    similarity: Number((1 - distance).toFixed(3)),
  }));

  console.table(results);
  engine.free();
}

main();

React Component

import { useEffect, useState } from 'react';
import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Everyday commuter backpack",
    "text": "Minimal black backpack with laptop sleeve and weather-resistant fabric.",
    "meta": "bags"
  },
  {
    "title": "Trail hydration pack",
    "text": "Running-focused pack with chest straps and soft flask pockets.",
    "meta": "bags"
  },
  {
    "title": "Weekend duffel",
    "text": "Large carry-all bag for short trips and gym gear.",
    "meta": "bags"
  },
  {
    "title": "Leather briefcase",
    "text": "Structured office bag with premium materials and organizer pockets.",
    "meta": "bags"
  },
  {
    "title": "Packable sling",
    "text": "Small cross-body bag for essentials during city travel.",
    "meta": "bags"
  },
  {
    "title": "Camera cube insert",
    "text": "Protective insert for carrying lenses inside a larger backpack.",
    "meta": "accessories"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

export function SimilarItemsExample() {
  const [engine, setEngine] = useState(null);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  useEffect(() => {
    let cancelled = false;
    let instance;

    (async () => {
      await init();
      const flat = new Float32Array(
        records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
      );
      instance = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
      if (!cancelled) setEngine(instance);
    })();

    return () => {
      cancelled = true;
      instance?.free();
    };
  }, []);

  useEffect(() => {
    if (!engine || query.trim().length < 2) {
      setResults([]);
      return;
    }

    const hits = JSON.parse(engine.search(new Float32Array(embedText(query)), 5));
    setResults(
      hits.map(([id, distance]) => ({
        ...records[id],
        similarity: Number((1 - distance).toFixed(3)),
      }))
    );
  }, [engine, query]);

  return (
    <section>
      <input
        value={query}
        onChange={(event) => setQuery(event.target.value)}
        placeholder="Describe the selected item"
      />
      <ul>
        {results.map((result) => (
          <li key={result.title}>
            <strong>{result.title}</strong> — {result.meta} (score {result.similarity})
          </li>
        ))}
      </ul>
    </section>
  );
}

How It Works

The flow is the same pattern you would use in production: turn each record into an embedding vector, build an HNSW graph with WasmSearchEngine.from_vectors(), and query it with the embedding for the current search text. HNSW keeps approximate nearest-neighbor search fast by traversing a layered graph instead of scanning every vector linearly.

Similarity scoring here is derived from the returned distance with 1 - distance so the UI can show an intuitive confidence number. If you swap in a real embedding model such as MiniLM, multilingual-e5, or precomputed server-generated vectors, the surrounding indexing and querying code stays almost identical.

When to Use This Pattern

  • More-like-this modules on product pages
  • Related reading blocks in content products
  • Media asset browsers
  • Shopping or inspiration apps with lightweight catalogs

Limitations

Similarity alone does not account for inventory, margin, or personalization. Large catalogs and business rules often require a server-side ranking layer on top of semantic nearest neighbors.
