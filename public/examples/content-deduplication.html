Build Content Deduplication with altor-vec

What you'll build: A deduplication pass that compares a draft item against an existing corpus and surfaces near-duplicates.

Bundle cost: 54KB · Query time: <1ms · Backend: None required

Install

npm install altor-vec

The Concept

Content teams accidentally create duplicates because titles differ even when the meaning is the same. A vector index helps you catch semantic overlap early so you can merge, redirect, or rewrite before publishing another near-identical page.

Browser-side dedupe is handy inside CMS tooling and editorial workflows. Editors can paste a draft title or summary and immediately see overlapping articles without waiting on an external service.

The example below indexes existing pages and then queries the draft summary against them. It is intentionally simple but runnable, which makes it a good starting point for editorial QA tooling.

This page uses a deterministic embedding helper so the snippets are runnable with only altor-vec. For production ranking quality, replace that helper with a real embedding model or precomputed vectors.

Full Code Example

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Invoice storage policy",
    "text": "How long finance records and customer invoices must be retained.",
    "meta": "help-center"
  },
  {
    "title": "Password reset guide",
    "text": "Steps for account recovery when a user cannot log in.",
    "meta": "help-center"
  },
  {
    "title": "Vendor security review",
    "text": "Checklist for onboarding new service providers securely.",
    "meta": "operations"
  },
  {
    "title": "Billing card updates",
    "text": "Instructions for changing the payment method on an account.",
    "meta": "help-center"
  },
  {
    "title": "Data deletion timeline",
    "text": "Explains deletion SLAs for churned customer data.",
    "meta": "privacy"
  },
  {
    "title": "Account export guide",
    "text": "How users can export analytics reports and billing history.",
    "meta": "help-center"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

async function main() {
  await init();

  const flat = new Float32Array(
    records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
  );

  const engine = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
  const hits = JSON.parse(engine.search(new Float32Array(embedText('new invoice retention article')), 4));

  const results = hits.map(([id, distance]) => ({
    ...records[id],
    similarity: Number((1 - distance).toFixed(3)),
  }));

  console.table(results);
  engine.free();
}

main();

import init, { WasmSearchEngine } from 'altor-vec';

const dims = 8;
const records = [
  {
    "title": "Invoice storage policy",
    "text": "How long finance records and customer invoices must be retained.",
    "meta": "help-center"
  },
  {
    "title": "Password reset guide",
    "text": "Steps for account recovery when a user cannot log in.",
    "meta": "help-center"
  },
  {
    "title": "Vendor security review",
    "text": "Checklist for onboarding new service providers securely.",
    "meta": "operations"
  },
  {
    "title": "Billing card updates",
    "text": "Instructions for changing the payment method on an account.",
    "meta": "help-center"
  },
  {
    "title": "Data deletion timeline",
    "text": "Explains deletion SLAs for churned customer data.",
    "meta": "privacy"
  },
  {
    "title": "Account export guide",
    "text": "How users can export analytics reports and billing history.",
    "meta": "help-center"
  }
];

function embedText(text) {
  const vector = new Float32Array(dims);
  for (const token of text.toLowerCase().split(/[^a-z0-9]+/).filter(Boolean)) {
    let hash = 2166136261;
    for (const char of token) hash = Math.imul(hash ^ char.charCodeAt(0), 16777619);
    const slot = Math.abs(hash) % dims;
    vector[slot] += 1;
    vector[(slot + token.length) % dims] += token.length / 10;
  }
  const magnitude = Math.hypot(...vector) || 1;
  return Array.from(vector, (value) => value / magnitude);
}

export function ContentDeduplicationExample() {
  const [engine, setEngine] = useState(null);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  useEffect(() => {
    let cancelled = false;
    let instance;

    (async () => {
      await init();
      const flat = new Float32Array(
        records.flatMap((record) => embedText(`${record.title} ${record.text} ${record.meta}`))
      );
      instance = WasmSearchEngine.from_vectors(flat, dims, 16, 200, 64);
      if (!cancelled) setEngine(instance);
    })();

    return () => {
      cancelled = true;
      instance?.free();
    };
  }, []);

  useEffect(() => {
    if (!engine || query.trim().length < 2) {
      setResults([]);
      return;
    }

    const hits = JSON.parse(engine.search(new Float32Array(embedText(query)), 5));
    setResults(
      hits.map(([id, distance]) => ({
        ...records[id],
        similarity: Number((1 - distance).toFixed(3)),
      }))
    );
  }, [engine, query]);

  return (
    <section>
      <input
        value={query}
        onChange={(event) => setQuery(event.target.value)}
        placeholder="Describe the draft content"
      />
      <ul>
        {results.map((result) => (
          <li key={result.title}>
            <strong>{result.title}</strong> — {result.meta} (score {result.similarity})
          </li>
        ))}
      </ul>
    </section>
  );
}

How It Works

The flow is the same pattern you would use in production: turn each record into an embedding vector, build an HNSW graph with WasmSearchEngine.from_vectors(), and query it with the embedding for the current search text. HNSW keeps approximate nearest-neighbor search fast by traversing a layered graph instead of scanning every vector linearly.

Similarity scoring here is derived from the returned distance with 1 - distance so the UI can show an intuitive confidence number. If you swap in a real embedding model such as MiniLM, multilingual-e5, or precomputed server-generated vectors, the surrounding indexing and querying code stays almost identical.

When to Use This Pattern

  • CMS authoring workflows
  • Knowledge base maintenance
  • Support article reviews before publishing
  • Dataset hygiene checks for RAG corpora

Limitations

Deduplication thresholds are subjective, and good results require consistent embeddings plus editorial review. Exact duplicate detection, canonical URL decisions, and large corpus management often need backend batch jobs.
